<script>

let stock = JSON.parse(localStorage.getItem("stock")) || [];
let income = parseInt(localStorage.getItem("income")) || 0;

function saveData() {
  localStorage.setItem("stock", JSON.stringify(stock));
  localStorage.setItem("income", income);
}

function addProduct() {
  let name = nameInput.value;
  let price = parseInt(priceInput.value);
  let qty = parseInt(qtyInput.value);

  stock.push({name, price, qty});
  saveData();
  displayStock();
}

function displayStock() {
  stockTable.innerHTML = "";
  stock.forEach(p => {
    stockTable.innerHTML += `
      <tr>
        <td>${p.name}</td>
        <td>₹ ${p.price}</td>
        <td>${p.qty}</td>
      </tr>`;
  });
}

function sellProduct() {
  let name = sellName.value;
  let qty = parseInt(sellQty.value);

  let product = stock.find(p => p.name === name);

  if(product && product.qty >= qty){
    product.qty -= qty;
    let total = product.price * qty;
    income += total;

    bill.innerHTML = `
      <h3>GOlZAR FURNITURE SHOP</h3>
      <hr>
      Product: ${product.name}<br>
      Quantity: ${qty}<br>
      Price: ₹ ${product.price}<br>
      <hr>
      <b>Total: ₹ ${total}</b>
      <br><br>
      Thank you!
    `;
  } else {
    alert("Product not available");
  }

  incomeDisplay.innerText = income;
  saveData();
  displayStock();
}

function printBill() {
  let content = bill.innerHTML;
  let win = window.open('', '', 'height=600,width=400');
  win.document.write('<html><body>' + content + '</body></html>');
  win.document.close();
  win.print();
}

incomeDisplay.innerText = income;
displayStock();
